// Helper function to get authenticated user data from cookies
function getAuthData(request: NextRequest): AuthData | null {
  try {
    const authToken = request.cookies.get("auth_token")?.value;
    const authUser = request.cookies.get("auth_user")?.value;
    const authExpires = request.cookies.get("auth_expires")?.value;

    if (!authToken || !authUser || !authExpires) {
      return null;
    }

    // Check if token is expired
    const expiresAt = parseInt(authExpires);
    if (Date.now() >= expiresAt) {
      return null;
    }

    // Parse and validate user data
    const user = JSON.parse(authUser);
    if (!user || !user.id || !user.role || !Array.isArray(user.role)) {
      return null;
    }

    return {
      user,
      token: authToken,
      expiresAt,
    };
  } catch (error) {
    console.error("‚ùå Error parsing auth data in middleware:", error);
    return null;
  }
}

    response.headers.set("X-Attempted-Route", pathname);

    return response;
  }

  const response = NextResponse.next();
  response.headers.set("X-User-Role", userRole);
  response.headers.set("X-User-ID", authData.user.id.toString());

  return response;
}

# Applying the Design Patterns and Concepts of Leading Interactive Learning Platforms

The success of platforms like Brilliant, Udemy, Skillshare, Sana Labs, and Duolingo is not accidental; it is the result of meticulously engineered learning experiences that blend pedagogical science with engaging user interface design and powerful gamification logic.     *   **Strategy:** Adopt a clean, modern display. Avoid unnecessary sidebars, pop-ups, or complex navigation within the learning environment itself.

    *   **Strategy:** Use **micro-animations** and **sound effects** to celebrate correct answers and soften the blow of incorrect ones. Ensure progress indicators (like progress bars or lesson completion icons) are always visible.
*   **Consistent Visual Language:** A strong, consistent brand identity (e.g., Duolingo's friendly green and owl mascot) makes the app feel reliable and familiar.
    *   **Strategy:** Develop a style guide with a consistent color scheme, font usage, and tone of voice. If applicable, use a mascot or recurring visual motif to guide the user.

## 3. Gamification and Engagement Logic

Gamification is the engine that drives daily engagement and transforms learning from a chore into a habit.

| Concept | Platform Example | Implementation Detail | Strategy for Your Project |
| :--- | :--- | :--- | :--- |
| **Streaks and Habit Formation** | **Duolingo** | The "Streak" mechanic encourages users to return daily to avoid losing their progress. This leverages the psychological principle of loss aversion. | **Implement a Daily Goal:** Set a small, achievable daily goal (e.g., "Complete 1 lesson"). Use push notifications or email reminders to prompt users to maintain their streak. |
| **Points, Levels, and Badges** | **Duolingo, Brilliant** | Users earn Experience Points (XP) for completing tasks, level up, and unlock badges for achieving milestones (e.g., "Master of Basics"). | **Reward Effort:** Award points for every completed action. | **Leaderboards and Competition** | **Duolingo** | Users are placed into leagues and compete against others to earn the most XP in a week. This taps into extrinsic motivation. | **Foster Community:** If appropriate for your audience, introduce a leaderboard or a "challenge mode" where users can compete with friends or peers. Ensure the competition is friendly and focused on learning effort. |
| **Virtual Currency/Incentives** | **Duolingo** | Users earn "Gems" or "Lingots" that can be spent on virtual items, streak freezes, or bonus content. | **Create a Store:** Introduce a virtual currency earned through learning. Allow users to spend this currency on cosmetic items, access to bonus materials, or helpful in-app tools. |

By systematically integrating these proven design patterns‚Äîfrom the micro-interactions of gamification to the macro-structure of personalized, bite-sized content‚Äîyou can create a highly effective and engaging interactive learning experience.
